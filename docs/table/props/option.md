<script setup>
const option = {
    operationWidth: 160,
    operationBtnTitle: '',

    hideOperation: false, // 是否隐藏操作列

    hideOprationDetail: false, // 是否隐藏操作列中的详情按钮
    hideOprationDelete: false, // 是否隐藏操作列中的删除按钮
    hideMenuAdd: false, // 是否隐藏添加按钮
    hidePage: false, // 是否隐藏分页器

}
</script>
